\begin{description}
  \item[Notion de stabilité:] On dit d'une solution de matching $\mathit{S}$ qu'elle est stable ssi
  $$ \neg\exists (m;w),(m';w') \in \mathit{S},\ m\ préfère\ w'\ à\ w\ et\ w'\ préfère\ m\ à\ m'$$
\end{description}
